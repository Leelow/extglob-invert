An easy way to hook process.exit().

## Install

```
$ npm install --save extglob-invert
```

## Usage

```js
const invert = require('extglob-invert');

invert('/path/to/file')             // !(path/to/file)
invert('   !  !( !/path/to/file)')  // /path/to/file
invert('!**')                       // **
invert('!(!!(**))')                 // **

```
